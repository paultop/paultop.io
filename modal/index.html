Pipeline Architecture

import os
import sys
from modal import Image, App

cuda_image = (
    Image.debian_slim(python_version="3.10")
    .apt_install("libgl1", "libglib2.0-0", "git", "ffmpeg")
    .pip_install(
        "numpy==1.26.4",
        "opencv-python-headless==4.8.1.78",
        "torch==2.1.2",
        "torchvision==0.16.2",
        "huggingface_hub",
        "safetensors",
        "pillow"
    )
    .run_commands(
        "git clone https://github.com/DepthAnything/Depth-Anything-V2.git /root/Depth-Anything-V2"
    )
)

app = App("video-depth-pipeline", image=cuda_image)

@app.function(
    gpu="A10G",              
    timeout=600,             
)
def process_video_depth(video_bytes: bytes) -> bytes:
    import cv2
    import torch
    import numpy as np
    
    sys.path.append('/root/Depth-Anything-V2')
    from depth_anything_v2.dpt import DepthAnythingV2

    input_path = "input_temp.mp4"
    temp_output_path = "raw_output.mp4"
    final_output_path = "web_ready_output.mp4"
    
    with open(input_path, "wb") as f:
        f.write(video_bytes)

    print("⏳ Loading Depth-Anything-V2-Small weights onto GPU...")
    from huggingface_hub import hf_hub_download
    model_path = hf_hub_download(repo_id="depth-anything/Depth-Anything-V2-Small", filename="depth_anything_v2_vits.pth")
    
    model_configs = {'vits': {'encoder': 'vits', 'features': 64, 'out_channels': [48, 96, 192, 384]}}
    depth_anything = DepthAnythingV2(**model_configs['vits'])
    depth_anything.load_state_dict(torch.load(model_path, map_location='cpu'))
    depth_anything = depth_anything.to("cuda").eval()

    cap = cv2.VideoCapture(input_path)
    fps = cap.get(cv2.CAP_PROP_FPS)
    width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    out_width = width * 2
    
    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
    out = cv2.VideoWriter(temp_output_path, fourcc, fps, (out_width, height))

    while cap.isOpened():
        ret, frame = cap.read()
        if not ret:
            break
            
        frame_rgb = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        with torch.no_grad():
            depth = depth_anything.infer_image(frame_rgb)
            
        depth_normalized = ((depth - depth.min()) / (depth.max() - depth.min()) * 255).astype(np.uint8)
        depth_bgr = cv2.cvtColor(depth_normalized, cv2.COLOR_GRAY2BGR)
        side_by_side = np.hstack((frame, depth_bgr))
        out.write(side_by_side)

    cap.release()
    out.release()
    
    print("🛠️ Executing container level FFmpeg transcode...")
    ffmpeg_cmd = f"ffmpeg -y -i {temp_output_path} -vcodec libx264 -pix_fmt yuv420p {final_output_path}"
    os.system(ffmpeg_cmd)

    with open(final_output_path, "rb") as f:
        return f.read()
